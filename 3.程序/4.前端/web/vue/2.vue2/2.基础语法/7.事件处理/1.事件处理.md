# 事件处理

`v-on:` 处理各个DOM事件

例子：

```javascript
// 表达式
<button v-on:click="counter += 1">{{counter}}</button> 

//直接调用方法
<button v-on:click="great"></button> 

 //可以传参
<button v-on:click="message('msg')"></button>

//如果需要访问原始DOM事件，使用 $event 传入
<button v-on:click="warn('msg',$event)"></button> 
new Vue({
    data:{
        counter:0
    }
    methods:{
        great:function(event){ //这里event接收的是原始的dom事件，默认传入，可以不写。

        },
        message:function(msg){
            alert(msg)
        },
        wart:function(msg,event){ //这里event接收的是原始的dom事件
            if(event){
                event.preventDefault()
            }
        }
    }
})
```
